\def \zigzag(#1,#2,#3) {
  \def\coord{{#1}}
  \def\n{#2}
  \def\x{#3}
  \foreach \xi in {0,...,\n} {
      \pgfmathtruncatemacro{\xj}{\xi+1};
      \pgfmathtruncatemacro{\xa}{\xi+\x};
      \pgfmathtruncatemacro{\xb}{\xa+1};
    
      \draw[-,line width=0.6mm](\xa,\coord[\xi])--(\xb,\coord[\xj]); 
  }
}
